private variable
  A : Set _

-- A list membership test.

member? :
  ((x y : A) → Maybe (x ≡ y)) →
  ((x : A) (xs : List A) → Maybe (x ∈ xs))
member? _   _ []       = nothing
member? _≟_ x (y ∷ xs) with x ≟ y
… | just eq = just (here eq)
… | nothing = there <$> member? _≟_ x xs
