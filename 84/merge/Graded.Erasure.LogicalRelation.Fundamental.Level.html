opaque

  -- Validity for Level.

  Levelʳ :
    ts » Γ ⊢ t ∷Level →
    γ ▸ Γ ⊩ʳ Level ∷[ m ∣ n ] U t
  Levelʳ ⊢t =
    ▸⊩ʳ∷⇔ .proj₂ λ ⊢σ _ →
    ®∷→®∷◂ $
    ®∷U⇔ .proj₂
      ( subst-⊢∷L ⊢t ⊢σ
      , U/Levelᵣ (λ { PE.refl → T.refl })
      )

    ▸⊩ʳ∷⇔ .proj₂ λ _ _ →
    ®∷→®∷◂ $
    ®∷Level⇔ .proj₂ (ok , U/Levelᵣ (λ { PE.refl → T.refl }))
