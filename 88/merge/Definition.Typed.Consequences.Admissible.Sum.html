  Sum-cong :
    Γ ⊢ a₁ ≡ a₂ ∷Level →
    Γ ⊢ b₁ ≡ b₂ ∷Level →
    Γ ⊢ A₁ ≡ A₂ ∷ U a₁ →
    Γ ⊢ B₁ ≡ B₂ ∷ U b₁ →
    Γ ⊢ Sum a₁ b₁ A₁ B₁ ≡ Sum a₂ b₂ A₂ B₂ ∷ U (a₁ supᵘₗ b₁)

    let ⊢a₁ , ⊢a₂ = wf-⊢≡∷L a₁≡a₂

  -- A typing rule for Sum.

  ⊢Sum∷U :
    Γ ⊢ a ∷Level →
    Γ ⊢ b ∷Level →
    Γ ⊢ A ∷ U a →
    Γ ⊢ B ∷ U b →
    Γ ⊢ Sum a b A B ∷ U (a supᵘₗ b)
  ⊢Sum∷U ⊢a ⊢b ⊢A ⊢B =
    syntacticEqTerm
      (Sum-cong (refl-⊢≡∷L ⊢a) (refl-⊢≡∷L ⊢b) (refl ⊢A) (refl ⊢B))
      .proj₂ .proj₁

opaque

  -- A typing rule for Sum.

  ⊢Sum :
    Γ ⊢ a ∷Level →
    Γ ⊢ b ∷Level →
    Γ ⊢ A ∷ U a →
    Γ ⊢ B ∷ U b →
    Γ ⊢ Sum a b A B
  ⊢Sum ⊢a ⊢b ⊢A ⊢B =
    univ (⊢Sum∷U ⊢a ⊢b ⊢A ⊢B)
