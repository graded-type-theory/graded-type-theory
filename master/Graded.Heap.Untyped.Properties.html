opaque

  lookup-det : {H : Heap k m} {t : Term n} {u : Term nвҖІ}

    case -вүЎ-functional p-рқҹҷвүЎq p-рқҹҷвүЎqвҖІ of О» {

    refl , refl , refl , refl }

    refl , refl , refl , refl }
