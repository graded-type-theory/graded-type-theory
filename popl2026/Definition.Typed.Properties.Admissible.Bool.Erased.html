-- Typing rules for Bool, true and false

opaque
  unfolding Bool

  -- A typing rule for Bool.

  ⊢Bool∷U :
    ⊢ Γ →
    Γ ⊢ Bool ∷ U zeroᵘ
  ⊢Bool∷U ⊢Γ =
    ΠΣⱼ (⊢zeroᵘ ⊢Γ) (ℕⱼ ⊢Γ)
     (Erasedⱼ-U Erased-ok (⊢OK∷U (var₀ (⊢ℕ ⊢Γ))))
     Σ-ok

opaque

  -- A typing rule for Bool.

  ⊢Bool :
    ⊢ Γ →
    Γ ⊢ Bool
  ⊢Bool = univ ∘→ ⊢Bool∷U

opaque
  unfolding Bool true

  -- A typing rule for true.

  ⊢true :
    ⊢ Γ →
    Γ ⊢ true ∷ Bool
  ⊢true ⊢Γ =
    prodⱼ (⊢Erased-OK (var₀ (⊢ℕ ⊢Γ)))
      (sucⱼ (zeroⱼ ⊢Γ))
      (PE.subst (_⊢_∷_ _ _)
         (PE.sym $
          PE.trans Erased-[] $
          PE.cong (Erased _) B.OK-[]) $
       []ⱼ Erased-ok (⊢zeroᵘ ⊢Γ) $
       _⊢_∷_.conv (starⱼ ⊢Γ Unitʷ-ok) (sym (OK-1≡ ⊢Γ)))
      Σ-ok
    where
    open TyR

opaque
  unfolding Bool false

  -- A typing rule for false.

  ⊢false :
    ⊢ Γ →
    Γ ⊢ false ∷ Bool
  ⊢false ⊢Γ =
    prodⱼ (⊢Erased-OK (var₀ (⊢ℕ ⊢Γ))) (zeroⱼ ⊢Γ)
      (PE.subst (_⊢_∷_ _ _)
         (PE.sym $
          PE.trans Erased-[] $
          PE.cong (Erased _) B.OK-[]) $
       []ⱼ Erased-ok (⊢zeroᵘ ⊢Γ) $
       _⊢_∷_.conv (starⱼ ⊢Γ Unitʷ-ok) (sym (OK-0≡ ⊢Γ)))
      Σ-ok
    where
    open TyR

-- Typing rules for Target

opaque
  unfolding Bool Target

  -- An equality rule for Target.

  Target-cong :
    drop k Γ ∙ Bool ⊢ A₁ ≡ A₂ →
    Γ ⊢ t₁ ≡ t₂ ∷ ℕ →
    Γ ⊢ u₁ ≡ u₂ ∷ Erased zeroᵘ (OK t₁) →
    Γ ⊢ Target k A₁ t₁ u₁ ≡ Target k A₂ t₂ u₂
  Target-cong A₁≡A₂ t₁≡t₂ u₁≡u₂ =
    [][]↑-cong A₁≡A₂ $
    PE.subst (_⊢_≡_∷_ _ _ _)
      (PE.cong (ΠΣ⟨_⟩_,_▷_▹_ _ _ _ _) $
       PE.sym $
       PE.trans Erased-[] $
       PE.cong (Erased _) B.OK-[]) $
    prod-cong (⊢Erased-OK (var₀ (⊢ℕ (wfEqTerm t₁≡t₂)))) t₁≡t₂
      (PE.subst (_⊢_≡_∷_ _ _ _)
         (PE.sym $
          PE.trans Erased-[] $
          PE.cong (Erased _) B.OK-[])
         u₁≡u₂)
      Σ-ok

private opaque

  -- A variant of Target-cong.

  Target-cong′ :
    drop k Γ ∙ Bool ⊢ A₁ ≡ A₂ →
    Γ ⊢ t ∷ ℕ →
    Γ ⊢ u ∷ Erased zeroᵘ (OK t) →
    Γ ⊢ Target k A₁ t u ≡ Target k A₂ t u
  Target-cong′ A₁≡A₂ ⊢t ⊢u =
    Target-cong A₁≡A₂ (refl ⊢t) (refl ⊢u)

opaque

  -- A typing rule for Target.

  ⊢Target :
    drop k Γ ∙ Bool ⊢ A →
    Γ ⊢ t ∷ ℕ →
    Γ ⊢ u ∷ Erased zeroᵘ (OK t) →
    Γ ⊢ Target k A t u
  ⊢Target ⊢A ⊢t ⊢u =
    syntacticEq (Target-cong′ (refl ⊢A) ⊢t ⊢u) .proj₁

private
  module Boolrec
    (Π-ok : Π-allowed 𝟙 p)
    (Π-𝟙-𝟘-ok : Π-allowed 𝟙 𝟘)
    (Unitˢ-ok : Unitˢ-allowed)
    (A₁≡A₂ : Γ ∙ Bool ⊢ A₁ ≡ A₂)
    (t₁≡t₂ : Γ ⊢ t₁ ≡ t₂ ∷ A₁ [ true ]₀)
    (u₁≡u₂ : Γ ⊢ u₁ ≡ u₂ ∷ A₁ [ false ]₀)
    where

    opaque

      ⊢Γ : ⊢ Γ
      ⊢Γ = wfEqTerm t₁≡t₂

    opaque

      ⊢Unitʷ : Γ ⊢ Unitʷ
      ⊢Unitʷ = ⊢Unit ⊢Γ Unitʷ-ok

    opaque

      ⊢starʷ : Γ ⊢ starʷ ∷ Unitʷ
      ⊢starʷ = starⱼ ⊢Γ Unitʷ-ok

    opaque

      ⊢[starʷ] : Γ ⊢ [ starʷ ] ∷ Erased zeroᵘ Unitʷ
      ⊢[starʷ] = []ⱼ Erased-ok (⊢zeroᵘ ⊢Γ) ⊢starʷ

    opaque

      ⊢Erased-Unitʷ : Γ ⊢ Erased zeroᵘ Unitʷ
      ⊢Erased-Unitʷ = syntacticTerm ⊢[starʷ]

    opaque

      Π-lemma :
        drop k Δ PE.≡ Γ →
        Δ ∙ ℕ ⊢ t ∷ ℕ →
        Δ ∙ ℕ ⊢
          Π 𝟙 , p ▷ Erased zeroᵘ (OK t) ▹
            Target (2+ k) A₁ (wk1 t) (var x0) ≡
          Π 𝟙 , p ▷ Erased zeroᵘ (OK t) ▹
            Target (2+ k) A₂ (wk1 t) (var x0)
      Π-lemma PE.refl ⊢t =
        let ⊢OK = ⊢Erased-OK ⊢t in
        ΠΣ-cong (refl ⊢OK)
          (Target-cong′ A₁≡A₂ (wkTerm₁ ⊢OK ⊢t) $
           PE.subst (_⊢_∷_ _ _)
             (PE.trans wk-Erased $
              PE.cong (Erased _) B.wk-OK) $
           var₀ ⊢OK)
          Π-ok

    opaque

      Π-[]₀-lemma :
        Γ ⊢ t [ u ]₀ ∷ ℕ →
        Γ ⊢ OK (t [ u ]₀) ≡ Unitʷ →
        Γ ⊢
          (Π 𝟙 , p ▷ Erased zeroᵘ (OK t) ▹ Target 2 A₁ (wk1 t) (var x0))
            [ u ]₀ ≡
          Π 𝟙 , p ▷ Erased zeroᵘ Unitʷ ▹ Target 1 A₂ (wk1 (t [ u ]₀))
            (var x0)
      Π-[]₀-lemma {t} ⊢t[u]₀ OK-t[u]₀≡Unit =
        let ⊢OK = ⊢Erased-OK ⊢t[u]₀ in
        PE.subst (flip (_⊢_≡_ _) _)
          (PE.sym $
           PE.cong₂ (ΠΣ⟨_⟩_,_▷_▹_ _ _ _)
             (PE.trans Erased-[] $
              PE.cong (Erased _) B.OK-[])
             (PE.trans (Target-[₀⇑] 1) $
              PE.cong (flip (Target _ _) _) $
              wk1-liftSubst t)) $
        flip
          (ΠΣ-cong $
           Erased-cong Erased-ok (refl-⊢≡∷L (⊢zeroᵘ ⊢Γ)) OK-t[u]₀≡Unit)
          Π-ok $
        Target-cong′ A₁≡A₂ (wkTerm₁ ⊢OK ⊢t[u]₀) $
        PE.subst (_⊢_∷_ _ _)
          (PE.trans wk-Erased $
           PE.cong (Erased _) B.wk-OK) $
        var₀ ⊢OK

    opaque

      Target-lemma-0 :
        drop k Δ PE.≡ Γ →
        Δ ∙ B ⊢ t ∷ Erased zeroᵘ Unitʷ →
        Δ ∙ B ⊢
          Target (1+ k) A₁ zero t ≡
          Target (1+ k) A₂ zero t
      Target-lemma-0 PE.refl ⊢t =
        let ⊢Δ∙B = wfTerm ⊢t in
        Target-cong′ A₁≡A₂ (zeroⱼ ⊢Δ∙B)
          (conv ⊢t $ sym $
           Erased-cong Erased-ok (refl-⊢≡∷L (⊢zeroᵘ ⊢Δ∙B)) (OK-0≡ ⊢Δ∙B))

    opaque

      Target-lemma-1 :
        drop k Δ PE.≡ Γ →
        Δ ∙ B ⊢ t ∷ Erased zeroᵘ Unitʷ →
        Δ ∙ B ⊢
          Target (1+ k) A₁ (suc zero) t ≡
          Target (1+ k) A₂ (suc zero) t
      Target-lemma-1 PE.refl ⊢t =
        let ⊢Δ∙B = wfTerm ⊢t in

           Erased-cong Erased-ok (refl-⊢≡∷L (⊢zeroᵘ ⊢Δ∙B)) (OK-1≡ ⊢Δ∙B))

              PE.trans []-[] $ PE.sym wk-[])

    opaque

      unitrec-lemma-0 :
        drop k Δ PE.≡ Γ →
        ⊢ Δ →
        Δ ∙ Unitʷ ⊢
          unitrec 𝟘 𝟘 (Target (2+ k) A₁ zero [ var x0 ])
            (var x0) (wk[ 1+ k ]′ u₁) ≡
          unitrec 𝟘 𝟘 (Target (2+ k) A₂ zero [ var x0 ])
            (var x0) (wk[ 1+ k ]′ u₂) ∷
          Target (1+ k) A₁ zero (var x0) [ [ var x0 ] ]↑
      unitrec-lemma-0 ≡Γ ⊢Δ =
        let ⊢Unitʷ  = ⊢Unit ⊢Δ Unitʷ-ok
            ⊢Unitʷ′ = ⊢Unit (∙ ⊢Unitʷ) Unitʷ-ok

        PE.subst (_⊢_≡_∷_ _ _ _)
          (PE.trans (Target-[₀⇑] 0) $
           PE.trans (PE.cong (Target _ _ _) []-[]) $
           PE.sym $ Target-[↑⇑] 0) $
        unitrec-cong′
          (Target-lemma-0 ≡Γ $
           []ⱼ Erased-ok (⊢zeroᵘ (∙ ⊢Unitʷ′)) (var₀ ⊢Unitʷ′))

    opaque

      erasedrec-lemma-0 :

        Δ ⊢ B ≡ Erased zeroᵘ Unitʷ →
