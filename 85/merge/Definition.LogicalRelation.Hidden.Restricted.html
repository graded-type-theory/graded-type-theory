-- The type formers

opaque

  -- Reducible types.

  infix 4 _⊩⟨_⟩_

  _⊩⟨_⟩_ : Cons m n → Universe-level → Term n → Set a

    ⦃ inc : Var-included or-empty Γ .vars ⦄ → Γ L.⊩⟨ l ⟩ A

opaque
  unfolding _⊩⟨_⟩_ _⊩⟨_⟩_≡_

  -- Changing type levels for _⊩⟨_⟩_≡_.

  level-⊩≡ :
    Γ ⊩⟨ l ⟩ A →
    Γ ⊩⟨ l ⟩ B →
    Γ ⊩⟨ l′ ⟩ A ≡ B →
    Γ ⊩⟨ l ⟩ A ≡ B
  level-⊩≡ ⊩A ⊩B A≡B = H.level-⊩≡ ⊩A ⊩B A≡B

opaque
  unfolding _⊩⟨_⟩_ _⊩⟨_⟩_≡_∷_

  -- Changing type levels for _⊩⟨_⟩_≡_∷_.

  level-⊩≡∷ :
    Γ ⊩⟨ l ⟩ A →
    Γ ⊩⟨ l′ ⟩ t ≡ u ∷ A →
    Γ ⊩⟨ l ⟩ t ≡ u ∷ A
  level-⊩≡∷ ⊩A t≡u = H.level-⊩≡∷ ⊩A t≡u

opaque

  -- Changing type levels for _⊩⟨_⟩_∷_.

  level-⊩∷ :
    Γ ⊩⟨ l ⟩ A →
    Γ ⊩⟨ l′ ⟩ t ∷ A →
    Γ ⊩⟨ l ⟩ t ∷ A
  level-⊩∷ ⊩A =
    ⊩∷⇔⊩≡∷ .proj₂ ∘→ level-⊩≡∷ ⊩A ∘→ ⊩∷⇔⊩≡∷ .proj₁

------------------------------------------------------------------------
-- Weakening

opaque
  unfolding _⊩⟨_⟩_≡_

  -- Weakening for _⊩⟨_⟩_≡_.

  wk-⊩≡ :
    ∇ » ρ ∷ʷ Η ⊇ Δ → ∇ » Δ ⊩⟨ l ⟩ A ≡ B → ∇ » Η ⊩⟨ l ⟩ wk ρ A ≡ wk ρ B
  wk-⊩≡ Η⊇Δ A≡B =
    let Η⊇Δ = ∷ʷ⊇→∷ʷʳ⊇ Η⊇Δ in
    H.wk-⊩≡ Η⊇Δ $ A≡B ⦃ inc = wk-Var-included-or-empty→ Η⊇Δ ⦄

opaque

  -- Weakening for _⊩⟨_⟩_.

  wk-⊩ : ∇ » ρ ∷ʷ Η ⊇ Δ → ∇ » Δ ⊩⟨ l ⟩ A → ∇ » Η ⊩⟨ l ⟩ wk ρ A
  wk-⊩ Η⊇Δ = ⊩⇔⊩≡ .proj₂ ∘→ wk-⊩≡ Η⊇Δ ∘→ ⊩⇔⊩≡ .proj₁

opaque
  unfolding _⊩⟨_⟩_≡_∷_

  -- Weakening for _⊩⟨_⟩_≡_∷_.

  wk-⊩≡∷ :
    ∇ » ρ ∷ʷ Η ⊇ Δ → ∇ » Δ ⊩⟨ l ⟩ t ≡ u ∷ A →
    ∇ » Η ⊩⟨ l ⟩ wk ρ t ≡ wk ρ u ∷ wk ρ A
  wk-⊩≡∷ Η⊇Δ t≡u =
    let Η⊇Δ = ∷ʷ⊇→∷ʷʳ⊇ Η⊇Δ in
    H.wk-⊩≡∷ Η⊇Δ $ t≡u ⦃ inc = wk-Var-included-or-empty→ Η⊇Δ ⦄

opaque

  -- Weakening for _⊩⟨_⟩_∷_.

  wk-⊩∷ :
    ∇ » ρ ∷ʷ Η ⊇ Δ → ∇ » Δ ⊩⟨ l ⟩ t ∷ A → ∇ » Η ⊩⟨ l ⟩ wk ρ t ∷ wk ρ A
  wk-⊩∷ Η⊇Δ = ⊩∷⇔⊩≡∷ .proj₂ ∘→ wk-⊩≡∷ Η⊇Δ ∘→ ⊩∷⇔⊩≡∷ .proj₁

opaque
  unfolding _⊩⟨_⟩_≡_

  -- Weakening of the definition context for _⊩⟨_⟩_≡_.

  defn-wk-⊩≡ :
    » ∇′ ⊇ ∇ →
    ∇ » Δ ⊩⟨ l ⟩ A ≡ B →
    ∇′ » Δ ⊩⟨ l ⟩ A ≡ B
  defn-wk-⊩≡ ξ⊇ A≡B =
    with-inc-⊩≡ λ ⦃ inc ⦄ → (H.defn-wk-⊩≡ ξ⊇ (A≡B ⦃ inc ⦄))

opaque

  -- Weakening of the definition context for _⊩⟨_⟩_.

  defn-wk-⊩ : » ∇′ ⊇ ∇ → ∇ » Δ ⊩⟨ l ⟩ A → ∇′ » Δ ⊩⟨ l ⟩ A
  defn-wk-⊩ ξ⊇ = ⊩⇔⊩≡ .proj₂ ∘→ defn-wk-⊩≡ ξ⊇ ∘→ ⊩⇔⊩≡ .proj₁

opaque
  unfolding _⊩⟨_⟩_≡_∷_

  -- Weakening of the definition context for _⊩⟨_⟩_≡_∷_.

  defn-wk-⊩≡∷ :
    » ∇′ ⊇ ∇ →
    ∇ » Δ ⊩⟨ l ⟩ t ≡ u ∷ A →
    ∇′ » Δ ⊩⟨ l ⟩ t ≡ u ∷ A
  defn-wk-⊩≡∷ ξ⊇ t≡u =
    with-inc-⊩≡∷ λ ⦃ inc ⦄ → (H.defn-wk-⊩≡∷ ξ⊇ (t≡u ⦃ inc ⦄))

opaque

  -- Weakening of the definition context for _⊩⟨_⟩_∷_.

  defn-wk-⊩∷ :
    » ∇′ ⊇ ∇ →
    ∇ » Δ ⊩⟨ l ⟩ t ∷ A →
    ∇′ » Δ ⊩⟨ l ⟩ t ∷ A
  defn-wk-⊩∷ ξ⊇ = ⊩∷⇔⊩≡∷ .proj₂ ∘→ defn-wk-⊩≡∷ ξ⊇ ∘→ ⊩∷⇔⊩≡∷ .proj₁

-- Reduction

opaque
  unfolding _⊩⟨_⟩_ _⊩⟨_⟩_≡_

  -- A reduction lemma for _⊩⟨_⟩_.

  ⊩-⇒* : Γ ⊢ A ⇒* B → Γ ⊩⟨ l ⟩ A → Γ ⊩⟨ l ⟩ A ≡ B
  ⊩-⇒* A⇒*B ⊩A = H.⊩-⇒* A⇒*B ⊩A

  unfolding _⊩⟨_⟩_∷_ _⊩⟨_⟩_≡_∷_

  -- A reduction lemma for _⊩⟨_⟩_∷_.

  ⊩∷-⇒* :
    Γ ⊢ t ⇒* u ∷ A →
    Γ ⊩⟨ l ⟩ t ∷ A →
    Γ ⊩⟨ l ⟩ t ≡ u ∷ A
  ⊩∷-⇒* t⇒*u ⊩t = H.⊩∷-⇒* t⇒*u ⊩t

  -- An expansion lemma for _⊩⟨_⟩_∷_.

  ⊩∷-⇐* :
    Γ ⊢ t ⇒* u ∷ A →
    Γ ⊩⟨ l ⟩ u ∷ A →
    Γ ⊩⟨ l ⟩ t ≡ u ∷ A
  ⊩∷-⇐* t⇒*u ⊩u = H.⊩∷-⇐* t⇒*u ⊩u

-- Escape lemmas

opaque
  unfolding _⊩⟨_⟩_

  -- An escape lemma for _⊩⟨_⟩_.

  escape-⊩ :
    ⦃ inc : Var-included or-empty (Γ .vars) ⦄ →
    Γ ⊩⟨ l ⟩ A → Γ ⊢ A
  escape-⊩ ⊩A = H.escape-⊩ ⊩A

opaque
  unfolding _⊩⟨_⟩_∷_

  -- An escape lemma for _⊩⟨_⟩_∷_.

  escape-⊩∷ :
    ⦃ inc : Var-included or-empty (Γ .vars) ⦄ →
    Γ ⊩⟨ l ⟩ t ∷ A → Γ ⊢ t ∷ A
  escape-⊩∷ ⊩t = H.escape-⊩∷ ⊩t

opaque
  unfolding _⊩⟨_⟩_≡_

  -- An escape lemma for _⊩⟨_⟩_≡_.

  escape-⊩≡ :
    ⦃ inc : Var-included or-empty (Γ .vars) ⦄ →
    Γ ⊩⟨ l ⟩ A ≡ B → Γ ⊢ A ≅ B
  escape-⊩≡ A≡B = H.escape-⊩≡ A≡B

opaque
  unfolding _⊩⟨_⟩_≡_∷_

  -- An escape lemma for _⊩⟨_⟩_≡_∷_.

  escape-⊩≡∷ :
    ⦃ inc : Var-included or-empty (Γ .vars) ⦄ →
    Γ ⊩⟨ l ⟩ t ≡ u ∷ A → Γ ⊢ t ≅ u ∷ A
  escape-⊩≡∷ t≡u = H.escape-⊩≡∷ t≡u
